export const Template = ({ children, ...args }) => (
    <Carousel {...args}>
        {[...new Array(20)].map((_, index) => {
            const height = Math.floor(300 + Math.random() * (500 - 300));
            return (
                <img
                    key={index}
                    width={500}
                    height={height}
                    src={`https://picsum.photos/500/${height}?random=${index}`}
                />
            );
        })}
    </Carousel>
);
