import Masonry from '@madeinhaus/core/ui/Masonry';
import { Meta, Story, Canvas, ArgsTable } from '@storybook/addon-docs';

<Meta title="UI/Masonry" component={Masonry} />

export const Template = args => (
    <Masonry key={JSON.stringify(args)} {...args}>
        {[...new Array(20)].map((_, index) => {
            return (
                <img
                    key={index}
                    style={{ width: '100%', display: 'block' }}
                    src={`https://picsum.photos/500/${Math.floor(
                        300 + Math.random() * (500 - 300 + 1)
                    )}?random=${index}`}
                />
            );
        })}
    </Masonry>
);
